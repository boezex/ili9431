20 enum class ili9341_commands : uint8_t
21 {
22  sw_reset = 0x01, // perform software reset, no parameters
23  slp_in = 0x10, // go into sleep mode, no parameters
24  slp_out = 0x11, // go out of sleep mode, no parameters
25  ptl_on = 0x12, // partial on, no parameters
26  nor_on = 0x13, // normal on, no paramteters
27  inv_off = 0x20, // display inversion off, no parameters
28  inv_on = 0x21, // display inversion on, no parameters
29  gamma_set = 0x26, // gamma set, 1 parameter (correction setting)
30  disp_off = 0x28, // display off, no parameters
31  disp_on = 0x29, // display on, no parameters
32  ca_set = 0x2A, // column address set, 4 parameters: SC[15:8], SC[7:0], EC[15:8], EC[7:0]
33  pa_set = 0x2B, // page address set, 4 parameters: SP[15:8], SP[7:0], EP[15:8], EP[7:0]
34  ram_wr = 0x2C, // memory/ram write, 1 parameter: D[17:0] (gek, 2 bytes?)
35  ptl_ar = 0x30, // partial area, 4 parameters: SR[15:8], SR[7:0], ER[15:8], ER[7:0]
36  vscr_def = 0x33, // vertical scrolling definition, 6 parameters: TFA[15:8], TFA[7:0], VSA[15:8], VSA[7:0], BFA[15:8], BFA[7:0]
37  mad_ctl = 0x36, // memory access control, 1 parameter: D[7] = MY, D[6] = MX, D[5] = MV, D[4] = ML, D[3] = BGR, D[2] = MH, D[1:0] = doesn't care
38  vscr_sadd = 0x37, // vertical scrolling start adress, 2 parameters: VSP[15:8], VSP[7:0]
39  pix_fmt = 0x3A, // pixel format set, see datasheet for details
40  frm_ctl1 = 0xB1, // frame control (normal mode), see datasheet for details
41  frm_ctl2 = 0xB2, // frame control (idle mode), see datasheet for details
42  frm_ctl3 = 0xB3, // frame control (partial mode), see datasheet for details
43  inv_ctl = 0xB4, // display inversion control, 1 parameter: D[2] = NLA, D[1] = NLB, D[0] = NLC
44  dfunc_ctl = 0xB6, // display function control, 4 parameters, see datasheet for details
45  pw_ctl1 = 0xC0, // power control, see datasheet for details
46  pw_ctl2 = 0xC1, // power control, see datasheet for details
47  vm_ctl1 = 0xC5, // VCOM control, see datasheet for details
48  vm_ctl2 = 0xC7, // VCOM control, see datasheet for details
49  gmc_p = 0xE0, // gamma correction positive, see datasheet for details
50  gmc_n = 0xE1 // gamma correction negative, see datasheet for details
51 };
